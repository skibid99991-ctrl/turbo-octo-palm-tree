-- Floating Button
local RikyFloatingButton = Instance.new("ScreenGui")
local ImageButton = Instance.new("ImageButton")
local UICorner = Instance.new("UICorner")

RikyFloatingButton.Name = "RikyFloatingButton"
RikyFloatingButton.Parent = game:GetService("CoreGui")
RikyFloatingButton.ZIndexBehavior = Enum.ZIndexBehavior.Sibling

ImageButton.Parent = RikyFloatingButton
ImageButton.BackgroundColor3 = Color3.fromRGB(25, 25, 25)
ImageButton.BorderSizePixel = 0
ImageButton.Position = UDim2.new(0.02, 0, 0.4, 0)
ImageButton.Size = UDim2.new(0, 60, 0, 60)
ImageButton.Image = RikyHub.ImageID
ImageButton.Draggable = true

UICorner.Parent = ImageButton
UICorner.CornerRadius = UDim.new(0, 12)

-- Button Animation
local rotateTween = TweenService:Create(ImageButton, 
    TweenInfo.new(2, Enum.EasingStyle.Linear, Enum.EasingDirection.Out, -1), 
    {Rotation = 360}
)

ImageButton.MouseEnter:Connect(function()
    rotateTween:Play()
    ImageButton.Size = UDim2.new(0, 65, 0, 65)
end)

ImageButton.MouseLeave:Connect(function()
    rotateTween:Pause()
    ImageButton.Rotation = 0
    ImageButton.Size = UDim2.new(0, 60, 0, 60)
end)

ImageButton.MouseButton1Click:Connect(function()
    RikyWindow:Minimize()
end)
